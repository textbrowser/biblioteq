28#ifndef _BIBLIOTEQ_ANIMATED_TOOL_BUTTON_H_
29#define _BIBLIOTEQ_ANIMATED_TOOL_BUTTON_H_
30
31#include <QPropertyAnimation>
32#include <QToolButton>
33
34class biblioteq_animated_tool_button: public QToolButton
35{
36 Q_OBJECT
37 Q_PROPERTY(QColor color READ color WRITE setColor)
38
39 public:
40 biblioteq_animated_tool_button(QWidget *parent):QToolButton(parent)
41 {
42 m_animator.setPropertyName("color");
43 m_animator.setTargetObject(this);
44 m_negativeColor = QColor(255, 36, 0); // A sprite of green.
45 m_originalColor = palette().color(QPalette::Window);
46 m_positiveColor = QColor(0, 177, 64); // A sprite of red.
47 }
48
49 ~biblioteq_animated_tool_button()
50 {
51 m_animator.stop();
52 }
53
54 QColor color(void) const
55 {
56 return QColor(Qt::blue); // A guess.
57 }
58
59 void animate(const int milliseconds)
60 {
61 animatePositively(milliseconds);
62 }
63
64 void animateNegatively(const int milliseconds)
65 {
66 animation(m_negativeColor, milliseconds);
67 }
68
69 void animatePositively(const int milliseconds)
70 {
71 animation(m_positiveColor, milliseconds);
72 }
73
74 private:
75 QColor m_negativeColor;
76 QColor m_originalColor;
77 QColor m_positiveColor;
78 QPropertyAnimation m_animator;
79
80 void animation(const QColor &color, const int milliseconds)
81 {
82 if(milliseconds <= 0)
83 {
84 m_animator.stop();
85 return;
86 }
87
88 m_animator.stop();
89 m_animator.setDuration(milliseconds);
90 m_animator.setEasingCurve(QEasingCurve::OutCubic);
91 m_animator.setEndValue(m_originalColor);
92 m_animator.setLoopCount(1);
93 m_animator.setStartValue(color);
94 m_animator.start();
95 }
96
97 void setColor(const QColor &color)
98 {
99 setStyleSheet
100 (QString("background-color: rgb(%1, %2, %3); "
101 "border: none; "
102 "padding: 5px;").
103 arg(color.red()).arg(color.green()).arg(color.blue()));
104 }
105};
106
107#endif
